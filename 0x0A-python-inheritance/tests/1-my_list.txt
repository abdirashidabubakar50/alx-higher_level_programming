# A subclass of list with an additional method to print the list in sorted order
>>> MyList = __import__('1-my_list').MyList
>>> my_list = MyList([3, 1, 4, 2, 5])
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> my_list = MyList([10, -1, 3, 7, 0])
>>> my_list.print_sorted()
[-1, 0, 3, 7, 10]
>>> my_list.print_sorted()
[-1, 0, 3, 7, 10]
>>> my_list = MyList([])
>>> my_list.print_sorted()
[]
>>>
